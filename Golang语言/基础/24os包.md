## os 模块

> os : 实现了操作系统的编程接口



### 1. 文件

```go
// 创建文件

		fmt.Println("创建失败")
```

```go
// 创建目录
```

```go
// 打开文件
func openCloes() {
	// 打开文件
	f, err := os.Open("b.txt")
	if err != nil {
		fmt.Printf("err: %v\n", err)
	} else {
		fmt.Printf("f.Name(): %v\n", f.Name())

		f.Close()

		fmt.Printf("err2: %v\n", err2)
	} else {
		fmt.Printf("f2.Name(): %v\n", f2.Name())
		f2.Close()
	}
```
```go
}
```

```go
// 读文件
func ReadFile() {

	// 打开文件
	f, _ := os.Open("bb.txt")

	for {
		// 建立buff 存字符
		buf := make([]byte, 5)
		n, err := f.Read(buf)
		// 读取到最后 才退出循环
		if err == io.EOF {
			break
		}
		fmt.Printf("n: %v\n", n)
		// 装换为字符
		fmt.Printf("string(buf): %v\n", string(buf))
	}
	f.Close()

	// 2. ReadAt()
	f2, _ := os.Open("bb.txt")
	buf := make([]byte, 10)
	// 从5的位置开始读
	n, _ := f2.ReadAt(buf, 5)
	fmt.Printf("n: %v\n", n)
	fmt.Printf("string(n): %v\n", string(buf))

	f2.Close()
}
```

### 3.File 文件写操作

> 写入到文件

```go
// 字节数组写入文件
func write() {

	// open只读权限   os.O_RDWR 可读可写 os.O_APPEND 文件后面追加 O_CREATE 覆盖 O_CREATE 没有文件创建
	f, _ := os.OpenFile("bbb.txt", os.O_RDWR|os.O_APPEND|os.O_CREATE, 755)

	// 写入 字节数组
	f.Write([]byte(" go go go"))
	f.Close()
}

// 字符串直接写入
func writeString() {
	f, _ := os.OpenFile("bb.txt", os.O_RDWR|os.O_TRUNC, 755)

	// 直接写入字符串
	f.WriteString("golang")
	f.Close()
}

// 某个位置写入
func writeAt() {
	f, _ := os.OpenFile("bb.txt", os.O_RDWR, 755)

	// 4的位置写入  aaa的字节数组
	f.WriteAt([]byte("aaa"), 4)
	f.Close()
}
```



### 4. 进程相关

```go
func main() {

	// 获得在运行的进程id
	fmt.Printf("os.Getpid: %v\n", os.Getpid)

	// 父id
	fmt.Printf("os.Getppid(): %v\n", os.Getppid())

	// 设置新的进程

	attr := &os.ProcAttr{

		// files 新进程的活动文件对象  标准输入 标准输出 标准错误输出
		Files: []*os.File{os.Stdin, os.Stdout, os.Stderr},
		// 新进程的环境变量
		Env: os.Environ(),
	}
	// 开启新进程   notepad 打开 a.txt
	p, err := os.StartProcess("C:\\Windows\\notepad.exe", []string{"C:\\Windows\\notepad.exe", "a.txt"}, attr)
	if err != nil {
		fmt.Printf("err: %v\n", err)
	}

	fmt.Printf("p.Pid: %v\n", p.Pid)

	// 通过进程id查找进程
	p2, _ := os.FindProcess(p.Pid)

	fmt.Printf("p2: %v\n", p2)

	// 等待5s执行函数
	time.AfterFunc(time.Second*5, func() {

		// 向 p进程发送退出信号
		p.Signal(os.Kill)
	})

	// 等待进程退出，返回进程状态
	ps, _ := p.Wait()
	fmt.Printf("ps: %v\n", ps)
}

```
