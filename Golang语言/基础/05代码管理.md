```go

1. go mod init hi  # 初始化

2. 创建一个新的包

mkdir hello 

touch hello.go

package hello

func Hello() string {

	return "hello golang"}

3. 主包引用其他包

package main

import (

	"fmt"

	"hi/hello" // 初始化名称+包名

)

func main() {

	s := hello.Hello()  // hello包的Hello函数

	fmt.Printf("s: %v\n", s)

}
```
