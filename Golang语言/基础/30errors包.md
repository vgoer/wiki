## errors包

> 实现了操作错误的函数

```go
func check(s string) (string, error) {

	if s == "" {
		// 新建错误
		err := errors.New("字符串不能为空")
		return "", err
	} else {
		return s, nil
	}
}
func main() {
	s, err := check("")
	if err != nil {
		fmt.Printf("err: %v\n", err.Error())
	} else {
		fmt.Printf("s: %v\n", s)
	}
}
```
