## 并发编程

> Go语言在语言层面天生支持并发，充分利用现代CPU的多核优势
>
> 并发： 同一时间执行多个任务
>
> 一些基本概念：[up](https://www.liwenzhou.com/posts/Go/concurrence/)



### 1. 协程

> 协程（coroutine）：非操作系统提供而是由用户自行创建和控制的用户态‘线程’，比线程更轻量级。` 看不懂也没关系，我也看不懂`

```go
// 创建一个协程  函数前加go关键字
go goadd()
// 创建多个协程  就有多个人干活
```

```go
func showMsg(msg string) {

	for i := 0; i < 10; i++ {
		fmt.Printf("msg: %v\n", msg)
		// 睡眠   Millisecond 毫秒
		time.Sleep(time.Millisecond * 100)
	}

}

// 只有一个主函数， 一个协程 ， 按照顺序执行
func main() {
	showMsg("golang")  // 打印完golang 才会打印java
	showMsg("java")
}

// 两个协程同时进行 
func main() {
	// 创建的协程
	go showMsg("golang")
    // main的协程
	showMsg("java")
}

// 主函数退出，程序结束了
func main() {
	// 主函数退出，程序结束了
	go showMsg("golang")
	go showMsg("java") // 不会打印任何东西

}
func main() {

	// 主函数退出，程序结束了
	go showMsg("golang")
	go showMsg("java") // 不会打印任何东西
	time.Sleep(time.Microsecond * 20000)
	fmt.Println("main end..")
}
```



### 2. 通道

> channel: ==不同的协程==之间通过通道==通信== 共享数据
>
> 有缓冲==同步==和无缓冲的==异步==

```go
// make 创建
var unbuff = make(chan string)   // 无缓存区通道
var buff = make(chan string, 10) // 10个缓冲区的通道
```



> 将值发送到通道 `<-`

```go
var str string = "我要去通道"
// 发送到通道
buff <- str

// 接收通道的值
data := <-buff
fmt.Printf("data: %v\n", data)
fmt.Printf("buff: %T\n", buff)
```

```go
// 协程之间传递值
//创建一个通道
var chanValue = make(chan int, 10)

func sendMsg() {
	// time.Now().Unix()时间戳
	rand.Seed(time.Now().Unix())

	val := rand.Intn(10)
	// 等待10秒
	time.Sleep(time.Second * 10)
	// 将值发送给通道
	chanValue <- val
}

func main() {

	// 最后关闭通道
	defer close(chanValue)

	go sendMsg()
	fmt.Println("wait ....")
	// 接收通道的值
	val2 := <-chanValue

	fmt.Printf("val2: %v\n", val2)
}
```



### 3. WaitGroup实现同步

> 同步： 协程之间还没执行完，你等下

```go
// 这样的不会打印所有协程 主协程已经结束了
func showMsg(i int) {
	fmt.Printf("i: %v\n", i)
}
func main() {

	for i := 0; i < 20; i++ {
		// 创建协程
		go showMsg(i)
	}
    
    // 主协程已经结束了
}
```

> 使用`sync包`实现同步

```go
// 每个协程都会执行
import (
	"fmt"
	"sync"
)

// 创建同步的waitGroup
var wp sync.WaitGroup

func showMsg(i int) {
	// or up.Done()
	defer wp.Add(-1)
	fmt.Printf("i: %v\n", i)
}
func main() {

	for i := 0; i < 20; i++ {
		// 创建协程
		go showMsg(i)

		// wp +1
		wp.Add(1)
	}

	wp.Wait() // 等待 wp==0
	fmt.Println("end...")
}
```



### 4. 通过Mutex实现同步

> 除了通过channel实现同步，还可以通过Mutex互斥锁实现同步

```go
// 等子协程进行完了都 执行主协程
var min int = 100

var wp sync.WaitGroup

// 互斥锁
var lock sync.Mutex

func add() {

	defer wp.Done()

	// 加锁
	lock.Lock()
	min += 1
	fmt.Printf("m++: %v\n", min)
	time.Sleep(time.Microsecond * 10)
	// 解锁
	lock.Unlock()
}

func sub() {
	defer wp.Done()
	// 加锁
	lock.Lock()
	min -= 1
	fmt.Printf("m--: %v\n", min)
	time.Sleep(time.Microsecond * 2)
	// 解锁
	lock.Unlock()

}

func main() {

	for i := 0; i < 100; i++ {
		wp.Add(1)
		go add()
		wp.Add(1)
		go sub()

	}

	// 等子协程完了，在执行主协程
	wp.Wait()
	fmt.Printf("min end ... : %v\n", min)
}

```



### 5. channel 遍历

> for + if   or  for range

```go
// 读取超过通道长度 而且没有关闭通道
// 创建有缓存区的通道
var clon = make(chan int, 10)
func main() {

	// 匿名函数和调用
	func() {
		for i := 0; i < 3; i++ {
			// 写入到通道中
			clon <- i
		}
	}()

	// 如果 读取超出通道的长度 而且没有关闭通道就会出现死锁   - deadlock!
	for i := 0; i < 4; i++ {

		r := <-clon

		fmt.Printf("r: %v\n", r)
	}
}
```

```go
// 循环通道
// 创建有缓存区的通道
var clon = make(chan int, 10)

func main() {

	// 匿名函数和调用
	func() {
		for i := 0; i < 3; i++ {
			// 写入到通道中
			clon <- i
		}
		// 关闭通道
		close(clon)
	}()
	
    //  1.0 
    for i := 0; i < len(clon); i++ {

     	r := <-clon

     	fmt.Printf("r: %v\n", r)
     }

	 // 2.0 一般使用for range
	 for v := range clon {
	 	fmt.Printf("v: %v\n", v)
	 }
	
    // 3.0
	for {
		v, ok := <-clon
		if ok {
			fmt.Printf("v: %v\n", v)
		} else {
			// 退出要
			break
		}
	}
}
```



### 6.Select 

> 类似switch语句，用来==控制通道里面的读写操作== 处理异步io操作

```go
select 中的case语句必须是一个channel操作
select 中的default 子句总是可以运行的
```

```go
var chanInt = make(chan int, 10)
var chanStr = make(chan string, 3)

func main() {
	func() {

		chanInt <- 100
		chanStr <- "hello go"

		defer close(chanInt)
		defer close(chanStr)
	}()

	// 死循环
	for {
		// select 语句
		select {
		case r := <-chanInt:
			fmt.Printf("int: %v\n", r)
		case r := <-chanStr:
			fmt.Printf("str: %v\n", r)
		default:
			fmt.Println("def 默认")
		}
		time.Sleep(time.Second)
	}

}
```





### 7. Timer

> 定时器的意思，实现一些定时操作
>
> 内部也是通道实现的

```go
//  NewTimer 实例化
timer := time.NewTimer(time.Second * 5)

// Now 当前时间
t1 := time.Now()
fmt.Printf("t1: %v\n", t1)

// 阻塞的 直到时间到了才执行
t2 := <-timer.C // t2 等待 timer 5秒
fmt.Printf("t2: %v\n", t2)
```

```go
<-timer.C  // 同上等待

time.Sleep(time.Second * 5)  // 同上一样
```

```go
// After() 等待多少秒后执行
t1 := time.Now()
fmt.Printf("t1: %v\n", t1)

<-time.After(time.Second * 3)
fmt.Printf("t1: %v\n", t1)
```

```go
// Stop() 停止阻塞
//  NewTimer 实例化
timer := time.NewTimer(time.Second * 4)

go func() {
    <-timer.C
    fmt.Println("timer ....")
}()

// 停止等待线程执行
s := timer.Stop()
if s {
    fmt.Println("timer stop")
}

// 主线程等待 6秒
time.Sleep(time.Second * 6)

fmt.Println("end ...")
```

```go
// Reset() 重置等待
timer := time.NewTimer(time.Second * 4) // 开始设置等待4秒

t1 := time.Now()
fmt.Printf("t1: %v\n", t1)

// 重置等待 1秒
timer.Reset(time.Second * 1)
<-timer.C
fmt.Println("end ...")
```



### 7.Ticker

> timer 只执行一次，Ticker 可以周期执行

```go
// 简单列子 一直执行

// 实例一个ticker
ticker := time.NewTicker(time.Second)
for _ = range ticker.C {
    fmt.Println("ticker... ")
}

// 停止
ticker.Stop()
```

```go
// 周期性 写入到通道中 
chanStr := make(chan string, 0)

ticker := time.NewTicker(time.Second)

go func() {
    for _ = range ticker.C {
        // select 写入到通道
        select {
            case chanStr <- "java":
        case chanStr <- "javascript":
        case chanStr <- "golang":
            }
    }
}()

sum := 0
for v := range chanStr {
    fmt.Printf("收到: %v\n", v)
    if sum >= 10 {
        ticker.Stop()
        break
    }
    sum += 2
}
```



### 8.原子变量

> 等线程执行完，才会执行主线程的程序

```go
// 互斥锁   你和你老板同时取钱， 100-10   90-10  你取没加锁，导致多取了10
var num int = 100

var lock sync.Mutex

func add() {
	lock.Lock()
	num++
	lock.Unlock()
}

func sub() {
	lock.Lock()
	num--
	lock.Unlock()
}

func main() {

	for i := 0; i < 100; i++ {
		go add()
		go sub()
	}

	fmt.Printf("num: %v\n", num)
	time.Sleep(time.Second * 2)
}
```

```go
// 另一个方法  原子变量 
func add() {
	// 比较 不一样+1
	atomic.AddInt32(&num, 1)
}

func sub() {
	atomic.AddInt32(&num, -1)
}

func main() {

	for i := 0; i < 100; i++ {
		go add()
		go sub()
	}

	time.Sleep(time.Second)
	fmt.Printf("num: %v\n", num)
}
```

```go
// 原子操作
var i int64 = 100

// 增加
atomic.AddInt64(&i, 2)
fmt.Printf("i: %v\n", i)

// 读
atomic.LoadInt64(&i)
fmt.Printf("i: %v\n", i)

// 写
atomic.StoreInt64(&i, 200)
fmt.Printf("i: %v\n", i)

//cas  比较值  赋值	 old 修改 new  返回bool
b := atomic.CompareAndSwapInt64(&i, 200, 500)
fmt.Printf("i: %v\n", i)
fmt.Printf("b: %v\n", b)
```
