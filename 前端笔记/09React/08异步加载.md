### 异步加载

> ==异步加载懒加载==：需要的时候才用到的资源

```jsx
// 这样资源都在一个bundle，太浪费资源
import { data } from './data';
function App(){

    const [text,setText] = React.useState('')

    const onChage = () => {
        setText(data)

    }
    return(
        <div>
            <h3>{text.name}</h3>
            <button onClick={onChage}>chage</button>
        </div>
    )
}
```

> 修改为

```jsx
// 没有和bundle 一起打包了
const [text,setText] = React.useState('')
const onChage = () => {

    // import 当函数用
    import('./data')
    .then((module) => {
        setText(module.data)
    })
}
```

> 如果是 `export default`暴露的要 `module.default`接受

> 异步加载方法

```js
// fun.js

export const getJson = (obj) => JSON.stringify(obj)

// app.js
function App(){

    const [text,setText] = React.useState('')
    const onChage = () => {
	
        import('./fun')
        .then((module) => {
            setText(module.getJson({name:'goer',age:10}))
        })
    }
    return(
        <div>
            <h3>{text}</h3>
            <button onClick={onChage}>chage</button>
        </div>
    )
}
```



### 1.异步组件

> 异步组件

```jsx
// MyAbout.jsx
export default function MyAbout(){
    return (
        <div>

            <h1>hello myAbout</h1>
        
        </div>
    )
}
```

```jsx

// lazy懒加载
const LazyComponent = React.lazy(() => import('./pages/MyAbout'));

function App(){
    return(
        <div>   
            {/* fallback 为加载是显示的组件 */}
            <React.Suspense fallback={<h2>Loading...</h2>}>
                <LazyComponent />
            </React.Suspense>
        </div>
    )
}
```
