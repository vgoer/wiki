### 组件开发

> 组件化开发： 页面拆分  
>
> 模块化开发： 功能拆分

```vue
// 创建多个页面组件
<script setup>
    // This starter template is using Vue 3 <script setup> SFCs
    // Check out https://vuejs.org/api/sfc-script-setup.html#script-setup
    import Head from './components/Head.vue'
    import Nav from './components/Nav.vue'
    import Content from './components/Content.vue'
</script>

<template>
<div class="head">
    <Head></Head>
    </div>
<div class="container">
    <div class="nav">
        <Nav></Nav>
    </div>
    <div class="content">
        <Content></Content>
    </div>
    </div>
</template>

<style>
    #app {
        font-family: Avenir, Helvetica, Arial, sans-serif;
        -webkit-font-smoothing: antialiased;
        -moz-osx-font-smoothing: grayscale;
    }
    .head{
        text-align: center;
        border: 1px solid blue;
    }
    .container{
        display: flex;
    }
    .nav{
        border: 1px solid red;
        height: 100vh;
        text-align: center;
        flex: 1;
    }
    .content{
        flex: 5;
        border:1px solid green;
    }
</style>
```

```vue
// 模块发开发  功能分开
// 新建 utils文件
```

```js
// sum 功能模块

function sum() {
    return 10+10
}

// 暴露接口
export default sum;
```

```vue
// 使用这个功能
<template>
    <h2>Content 主体部分</h2>
    <h5>sum值: {{number}}</h5>
</template>

<script>
import sum from '../utils/sum'

export default {

    // 计算属性
    computed:{
        number(){
            // 调用
            return sum();
        }
    }
}
</script>
```
