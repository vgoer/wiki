## Router

> 路由：实现==页面的跳转==

```shell
# 安装 router
npm install vue-router@4
```

```vue
// 实现一个点击导航 主体内容切换 

# 设置跳转标签
<template>
 	// 和html的a标签作用一样
    <router-link to="/">首页</router-link> | 
    <router-link to="/blog">博客</router-link>
	// 路由切换显示的东西放在 router-view
    <router-view></router-view>
</template>
```

```js
// 路由文件  router.js
import { createRouter,createWebHashHistory } from 'vue-router'

// 引入组件
import Home from './components/Home.vue'
import Blog from './components/Blog.vue'

// 创建路由
const router = createRouter({
    // 路由模式
    history:createWebHashHistory(),
    // 路由列表
    routers:[
        // 首页组件
        {
            path:'/',
            component:Home,
        },
        // 博客 
        {
            path:'/blog',
            component:Blog,
        },
    ]   
})
// 暴露出去
export default router;
```

```js
// mian.js 使用这个路由文件


// 引入 路由文件
import router from './router';

// .use() 使用
createApp(App).use(router).mount('#app');
```

> 路由懒加载： 好像可以改善性能

```js
import { createRouter, createWebHashHistory} from 'vue-router'

const routes =  [
    {path:'/', component:() => import('./components/Home.vue')},
    {path:'/blog', component:() => import('./components/Blog.vue')}
    {path:'/:pathMatch(.*)*',component:() => import('./404.vue')}
]

const router = createRouter({
    history:createWebHashHistory(),
    routes,
})

export default router
```



### 路由传参

> 传递参数：页面之间

```vue
// 实现 列表 到博客详情 切换功能  传递页面参数
```

```js
// 路由js 注意修改 接收参数 :id
import { createRouter, createWebHashHistory} from 'vue-router'


const routes =  [
    {path:'/', component:() => import('./components/List.vue')},
    //  :id   传递的参数路由也要设置格式
    {path:'/blog/:id', component:() => import('./components/Blog.vue')}
]


const router = createRouter({
    history:createWebHashHistory(),
    routes,
})

export default router
```

```vue
# 文章列表
<template>
    <h2>文章列表页</h2>
    <ul>
        <li v-for="list,index in blogList" :key="index">
        <!-- 传递参数 设置  拼接url -->
            <router-link :to="'/blog/' + list.id">
                {{list.id}}.{{list.title}}
            </router-link>
        </li>
    </ul>
</template>
<script>
    export default{
        data(){
            return{
                // 模拟后台数据
                blogList:[
                    {id:1,title:'html',cont:"html文章"},
                    {id:2,title:'css',cont:"css文章"},
                    {id:3,title:'js',cont:"js文章"},
                    {id:4,title:'vue',cont:"vue文章"},
                    {id:5,title:'ajax',cont:"ajax文章"},
                ]
            }
        }
    }

</script>


<style>
h2 {
    border: 1px solid red;
}
</style>
```

```vue
# 文章详情页  $toute暴露
<template>
    <h2>博客页id:{{$route.params.id}}</h2>

    <div class="con">
        {{blogCon}}
    </div>
</template>


<script>
    export default{
        data(){
            return{
                blogCon:''
            }
        },  
        // 生命周期获取数据
        beforeMount(){
            let id = this.$route.params.id;
            this.idGetDate(id);
        },
        methods:{
            // 模拟后台数据 通过id获取到
            idGetDate(id){
                switch (id){
                    case "1":this.blogCon = "HTML详情";break;
                    case "2":this.blogCon =  "CSS详情";break;
                    case "3":this.blogCon =  "JS详情";break;
                    case "4":this.blogCon =  "Vue详情";break;
                    case "5":this.blogCon =  "AJAX详情";break;
                }
            }
        }
    }

</script>

<style>
h2{
    border: 1px solid red;
}
</style>
```
