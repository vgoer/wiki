### ref 响应式

> vue3想要实现数据和视图同时修改需要使用 `ref 对象`

```vue
<script setup lang="ts">
    let msg: string = "goer hi!"

    function modMse() {
        msg = "我被修改了"
        console.log(msg);
        
    }
</script>

<template>
    <h2>{{msg}}</h2>

    <button @click="modMse">修改</button>
</template>

// 这样是没有响应式的，要使用ref对象

<script setup lang="ts">

    import {ref} from 'vue'
    let msg  = ref('goer')

    function modMse() {
        // ref.value 
        msg.value = "我被修改了"
        // ref对象 
        console.log(msg);
        
    }
</script>
```



### reactive响应式对象

> 对象实现响应式 `reactive`

```vue
<template>
    <h2>name:{{Cat.name}}</h2>
    <button @click="setName">修改name</button>
</template>


<script setup lang="ts">
    import {ref,reactive} from 'vue'
    // reactive 修改对象最佳方法
    const Cat = reactive({
        name:"小猫",
        age:2,
        love:'骨头',
    })
    let setName = () => {
        Cat.name = '小小白'
    }
</script>

// 修改数组
<template>
    <h3>{{arr}}</h3>
    <button @click="setValue">修改值</button>
</template>
<script setup lang="ts">
    import {ref,reactive} from 'vue'
    let arr = reactive(['aaa',1111,'bbb']);

    let setValue = () => {
        // 修改数组的相应是
        arr[1] = 44444
        console.log(arr[1])
    }
</script>
```
