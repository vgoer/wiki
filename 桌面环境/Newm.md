### 1. newm-atha

```shell
yay -S newm-atha-git

start-newm -d

#  配置文件 
cd
mkdir -p .config/newm
cp .local/lib/pythonX.Y/site-packages/newm/default_config.py .config/newm/config.py
vim .config/newm/config.py
```
